drop database  if  exists stusers;
create database stusers;
use stusers;
create table  user(
	id int primary key auto_increment,
	userId  varchar(20),
	password varchar(20),
	userName varchar(20)
	)Engine=InnoDB  default charset=utf8;
	
create table manger(
	id int primary key auto_increment,
	userId  varchar(20),
	password varchar(20),
	userName varchar(20)
	)Engine=InnoDB  default charset=utf8;
	
create table users(
	id int primary key auto_increment,
	stuId  varchar(20),
	stuName varchar(20),
	infoId  varchar(20),	
	major  varchar(20),
	school  varchar(20)
	)Engine=InnoDB  default charset=utf8;
	
create table info(
	id int primary key auto_increment,
	value  varchar(40)
